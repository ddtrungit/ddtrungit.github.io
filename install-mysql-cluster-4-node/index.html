Install mysql cluster 4 node

By Duong Dinh Trung

172.16.2.89        MysqlNode_01

172.16.2.90        ManagementNode

172.16.2.91        DataNode_02

172.16.2.93        DataNode_01

Cấu hình 1 Management node

# wget https://dev.mysql.com/get/Downloads/MySQL-Cluster-8.0/mysql-cluster-community-8.0.24-1.el7.x86_64.rpm-bundle.tar

# tar -xvf mysql-cluster-community-8.0.24-1.el7.x86_64.rpm-bundle.tar

# yum -y install epel-release

# yum install perl-Class-MethodMaker -y

# yum -y install perl-Data-Dumper

# yum -y remove mariadb-libs

# yum -y localinstall mysql-cluster-community-common-8.0.24-1.el7.x86_64.rpm mysql-cluster-community-libs-8.0.24-1.el7.x86_64.rpm mysql-cluster-community-client-plugins-8.0.24-1.el7.x86_64.rpm mysql-cluster-community-client-8.0.24-1.el7.x86_64.rpm mysql-cluster-community-management-server-8.0.24-1.el7.x86_64.rpm

# mkdir -p /var/lib/mysql-cluster

# vi /var/lib/mysql-cluster/config.ini

#TCP PARAMETERS
[tcp default]
SendBufferMemory=8M
ReceiveBufferMemory=8M
# Increasing the sizes of these 2 buffers beyond the default values
# helps prevent bottlenecks due to slow disk I/O. Default 2M

[shm default]
ShmSize=8M
ShmSpintime=200
SendBufferMemory=4M

# MANAGEMENT NODE PARAMETERS
[ndb_mgmd default]
# Directory for MGM node log files
DataDir=/var/lib/mysql-cluster

[ndb_mgmd]
#Management Node db1
#Id=1
HostName=172.16.2.90

[ndbd default]
NoOfReplicas=2 # Number of replicas
#DataMemory=256M # Memory allocate for data storage
#IndexMemory=128M # Memory allocate for index storage
#BackupDataDir=/var/lib/mysqlcluste-backup
#DataMemory=2G
#Directory for Data Node
DataDir=/var/lib/mysql-cluster

LockPagesInMainMemory=1
# On Linux and Solaris systems, setting this parameter locks data node
# processes into memory. Doing so prevents them from swapping to disk,
# which can severely degrade cluster performance.
DataMemory=2048M
# The value provided for DataMemory assumes 4 GB RAM
# per data node. However, for best results, you should first calculate
# the memory that would be used based on the data you actually plan to
# store (you may find the ndb_size.pl utility helpful in estimating
# this), then allow an extra 20% over the calculated values. Naturally,
# you should ensure that each data node host has at least as much
# physical memory as the sum of these two values.
# ODirect=1
# Enabling this parameter causes NDBCLUSTER to try using O_DIRECT
# writes for local checkpoints and redo logs; this can reduce load on
# CPUs. We recommend doing so when using NDB Cluster on systems running
# Linux kernel 2.6 or later.
#NoOfFragmentLogFiles=100
#MaxNoOfConcurrentOperations=1000
#MaxNoOfConcurrentTransactions=500
#SchedulerSpinTimer=400
#SchedulerExecutionTimer=100
#RealTimeScheduler=0
# Setting these parameters allows you to take advantage of real-time scheduling
# of NDB threads to achieve increased throughput when using ndbd. They
# are not needed when using ndbmtd; in particular, you should not set
# RealTimeScheduler for ndbmtd data nodes.
TimeBetweenGlobalCheckpoints=1000
TimeBetweenEpochs=200
RedoBuffer=32M
# CompressedLCP=1
# CompressedBackup=1
# Enabling CompressedLCP and CompressedBackup causes, respectively, local
#checkpoint files and backup files to be compressed, which can result in a space
#savings of up to 50% over noncompressed LCPs and backups.
# MaxNoOfLocalScans=64
MaxNoOfTables=1024
MaxNoOfOrderedIndexes=256

[ndbd]
#Data Node db1
#Id=2
HostName=172.16.2.93
#LockExecuteThreadToCPU=1
#LockMaintThreadsToCPU=0
# On systems with multiple CPUs, these parameters can be used to lock NDBCLUSTER
# threads to specific CPUs

[ndbd]
#Data Node db2
#Id=3
HostName=172.16.2.91
#LockExecuteThreadToCPU=1
#LockMaintThreadsToCPU=0
# On systems with multiple CPUs, these parameters can be used to lock NDBCLUSTER
# threads to specific CPUs

[mysqld]
#SQL Node db4
#Id=4
HostName=172.16.2.89
#ndb-wait-setup=120

Chạy lệnh này để khởi chạy Management node:

ndb_mgmd --initial --config-file=/var/lib/mysql-cluster/config.ini

ndb_mgm

Nếu không nhận config thì xóa file này đi rồi chạy lại:

/usr/mysql-cluster/ndb_1_config.bin.1

Cấu hình 1 SQL Node

# yum -y install epel-release

# yum install perl-Class-MethodMaker -y

# yum -y install perl-Data-Dumper

# yum -y remove mariadb-libs

# yum -y localinstall mysql-cluster-community-common-8.0.24-1.el7.x86_64.rpm mysql-cluster-community-libs-8.0.24-1.el7.x86_64.rpm mysql-cluster-community-client-plugins-8.0.24-1.el7.x86_64.rpm mysql-cluster-community-client-8.0.24-1.el7.x86_64.rpm mysql-cluster-community-server-8.0.24-1.el7.x86_64.rpm

vi /etc/my.cnf

[mysqld]

datadir=/var/lib/mysql

socket=/var/lib/mysql/mysql.sock

log-error=/var/log/mysqld.log

pid-file=/var/run/mysqld/mysqld.pid

ndbcluster

ndb-connectstring=172.16.2.90  # IP address of Management Node

default_storage_engine=ndbcluster # Define default Storage Engine used by MySQL

ndb-use-exact-count=0

ndb-index-stat-enable=0

ndb-force-send=1

optimizer-switch=engine_condition_pushdown=on

max_allowed_packet=1073741824

#ndb-default-column-format=DYNAMIC

innodb_strict_mode=0

innodb_log_file_size=256M

innodb_log_buffer_size=256M

[mysql_cluster]

ndb-connectstring=172.16.2.90  # IP address of Management Node

Khởi động sql node:

systemctl start mysqld

Chú ý: khi chạy lần đầu tiên, mysql server sẽ tạo một mật khẩu tạm và ghi vào file log tại vị trí /var/log/mysqld.log

mysql_secure_installation

ndb_mgm
